## ğŸ” How to Parse Tickers

### Python Code to Decode Any Ticker

```python
def parse_ticker(ticker: str) -> dict:
    """
    Parse any Kalshi ticker into components.

    Example:
        parse_ticker("KXNFLGAME-25SEP22DETBAL-BAL")
        Returns: {
            'sport': 'KXNFLGAME',
            'date': '25SEP22',
            'teams': 'DETBAL',
            'team1': 'DET',
            'team2': 'BAL',
            'outcome': 'BAL',
            'is_sports': True
        }
    """
    parts = ticker.split('-')

    if len(parts) < 2:
        return {'error': 'Invalid ticker format'}

    result = {
        'full_ticker': ticker,
        'prefix': parts[0],
        'is_sports': parts[0].startswith('KX')
    }

    if result['is_sports'] and len(parts) >= 2:
        # Sports ticker
        result['sport'] = parts[0]

        # Parse event part (date + teams)
        event = parts[1]

        # Extract date (first 7 characters: YYMMMDD)
        if len(event) >= 7:
            result['date'] = event[:7]
            result['teams'] = event[7:]

            # Try to split teams (usually 3-6 chars each)
            teams_str = event[7:]
            if len(teams_str) >= 6:
                # Assume equal length team codes
                mid = len(teams_str) // 2
                result['team1'] = teams_str[:mid]
                result['team2'] = teams_str[mid:]

        # Get outcome if present
        if len(parts) >= 3:
            result['outcome'] = parts[2]

    return result

# Test it
ticker = "KXNFLGAME-25SEP22DETBAL-BAL"
parsed = parse_ticker(ticker)
print(f"Sport: {parsed['sport']}")
print(f"Date: {parsed['date']}")
print(f"Teams: {parsed['team1']} vs {parsed['team2']}")
print(f"Winner: {parsed['outcome']}")
```

### Find Games by Date

```python
async def find_games_on_date(date_str: str):
    """
    Find all NFL games on a specific date.

    Args:
        date_str: Format "25SEP22" for Sept 22, 2025
    """
    source = KalshiMarketsSource(
        series_ticker="KXNFLGAME",
        status=None
    )

    async with source:
        async for df in source.collect():
            # Filter by date pattern in ticker
            games_on_date = df[
                df['ticker'].str.contains(date_str, na=False)
            ]

            if not games_on_date.empty:
                print(f"Games on {date_str}:")
                unique_events = games_on_date['event_ticker'].unique()
                for event in unique_events:
                    print(f"  {event}")

            return games_on_date
```
