## Your First Trading Bot

Let's create a simple bot that monitors NFL markets and alerts on opportunities:

```python
import asyncio
from neural_sdk import NeuralSDK

async def nfl_opportunity_bot():
    """Monitor NFL markets for trading opportunities"""
    
    # Initialize SDK
    sdk = NeuralSDK.from_env()
    websocket = sdk.create_websocket()
    
    # Track opportunities
    opportunities = []
    
    @websocket.on_market_data
    async def handle_market(data):
        # Look for undervalued markets
        if data.yes_price < 0.25 and data.volume > 10000:
            opportunities.append({
                'ticker': data.ticker,
                'price': data.yes_price,
                'volume': data.volume
            })
            print(f"ðŸŽ¯ Opportunity: {data.ticker}")
            print(f"   Price: ${data.yes_price:.2f}")
            print(f"   Volume: ${data.volume:,.0f}")
    
    # Connect and subscribe
    print("ðŸ”Œ Connecting to Kalshi WebSocket...")
    await websocket.connect()
    
    print("ðŸ“¡ Subscribing to NFL markets...")
    await websocket.subscribe_markets(['KXNFL*'])
    
    print("âœ… Bot is running! Press Ctrl+C to stop")
    
    try:
        await websocket.run_forever()
    except KeyboardInterrupt:
        print(f"\nðŸ“Š Found {len(opportunities)} opportunities")
        await websocket.disconnect()

# Run the bot
if __name__ == "__main__":
    asyncio.run(nfl_opportunity_bot())
```

## Common Patterns

### Event-Driven Strategy

```python
@websocket.on_market_data
async def momentum_strategy(data):
    """React to market momentum"""
    if data.price_change_1h > 0.10:  # 10% increase
        await sdk.place_order(
            ticker=data.ticker,
            side='YES',
            size=100
        )
```

### Scheduled Analysis

```python
import schedule

def analyze_markets():
    """Run analysis every hour"""
    markets = sdk.get_markets(sport='NFL')
    for market in markets:
        if market.volume < market.average_volume * 0.5:
            print(f"Low volume alert: {market.ticker}")

schedule.every().hour.do(analyze_markets)
```

### Risk Management

```python
from neural_sdk.trading import RiskManager

risk_manager = RiskManager(
    max_position_size=1000,
    max_daily_loss=500,
    position_limit=10
)

@websocket.on_market_data
async def safe_trading(data):
    if risk_manager.can_trade(data.ticker, 100):
        # Execute trade
        pass
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="ModuleNotFoundError">
    If you see `ModuleNotFoundError`, ensure you've installed the latest version:
    ```bash
    pip install --force-reinstall git+https://github.com/IntelIP/Neural-Trading-Platform.git@main
    ```
  </Accordion>
  
  <Accordion title="Authentication Failed">
    Check your `.env` file has the correct credentials:
    ```python
    import os
    print(f"API Key: {os.getenv('KALSHI_API_KEY_ID')}")
    print(f"Environment: {os.getenv('KALSHI_ENVIRONMENT')}")
    ```
  </Accordion>
  
  <Accordion title="WebSocket Connection Issues">
    Ensure you're using the production environment:
    ```python
    config = SDKConfig(
        environment="production",  # Not "demo" or "development"
        api_key_id=os.getenv("KALSHI_API_KEY_ID"),
        api_secret=os.getenv("KALSHI_API_SECRET")
    )
    ```
  </Accordion>
</AccordionGroup>
